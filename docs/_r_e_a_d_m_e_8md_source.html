9 ## Features
10 
11 * Read time
12 * Set time
13 * Read date and time
14 * Set date and time
15 * Read Unix Epoch UTC 32-bit timestamp
16 * Read temperature (0.25 degree resolution)
17 * Alarm 1 (second/minute/hour/day/date match)
18 * Alarm 2 (minute/hour/day/date match)
19 * Polling and Alarm ```INT/SQW``` interrupt pin
20 * Control ```32kHz``` out signal (enable/disable)
21 * Control ```SQW``` signal (disable/1/1024/4096/8192Hz)
22 * Configure aging offset
23 * Serial terminal interface
24 * Full RTC register access
25 * Easy debug functionality
26 * Basic and advanced examples
27 * Set date/time over serial with Python script
28 * Low RAM footprint:
29  * ```sizeof(DS3231)```: 1 Byte RTC object.
30  * ```sizeof(DS3231_DateTime)```: 8 Bytes date/time object.

73 ## Usage
74 
75 **Initialization**
76 
77 ```c++
78 #include <Wire.h>
79 #include <DS3231.h>
80 
81 // Create DS3231 RTC object
82 static DS3231 rtc;
83 
84 
85 void setup()
86 {
87  // Initialize TWI with a 100kHz (default) or 400kHz clock
88  Wire.begin();
89  Wire.setClock(400000);
90 
91  // Initialize RTC
92  while (rtc.begin()) {
93  // Error: Could not detect DS3231 RTC, retry after some time
94  delay(3000);
95  }
96 }
97 ```
98 
99 **Check oscillator status at startup**
100 
101 ```c++
102 // Check oscillator status
103 if (rtc.isOscillatorStopped()) {
104  // Error: DS3231 RTC oscillator stopped. Date/time cannot be trusted.
105  // Set new date/time before reading date/time.
106  while (1) {
107  ;
108  }
109 }
110 ```

198 **Program Alarm 1**
199 
200 Note: Alarm 1 and Alarm 2 have different behavior. Please refer to the documentation which ```Alarm1Type``` and ```Alarm2Type``` are supported. Some examples:
201 
202 ```c++
203 // Generate alarm 1 every second
204 rtc.setAlarm1(Alarm1EverySecond, 0, 0, 0, 0);
205 
206 // Generate alarm 1 every minute and second match
207 rtc.setAlarm1(Alarm1EverySecond, 0, 0, 45, 30);
208 
209 // Generate alarm 1 every day, hour, minute and second match
210 rtc.setAlarm1(Alarm1MatchDay,
211  1, // Alarm day match (1 = Monday)
212  12, // Alarm hour match
213  45, // Alarm minute match
214  30 // Alarm second match
215 );
216 ```

231 **Alarm polling**
232 
233 Note: The ```INT``` pin changes to low when an Alarm 1 or Alarm 2 match occurs and and the interrupt is enabled. The pin remains low until both alarm flags are cleared by the application.
234 
235 ```c++
236 // Poll alarm 1 flag
237 if (rtc.getAlarmFlag(Alarm1)) {
238  // Handle Alarm 1
239 
240  // Clear alarm 1 flag
241  rtc.clearAlarmFlag(Alarm1);
242 }
243 
244 // Poll alarm 2 flag
245 if (rtc.getAlarmFlag(Alarm2)) {
246  // Handle Alarm 2
247 
248  // Clear alarm 2 flag
249  rtc.clearAlarmFlag(Alarm2);
250 }
251 ```
252 
253 **Alarm interrupt**
254 
255 Note: Enabling interrupt will disable the ```SQW``` output signal.
256 
257 ```c++
258 // Uno, Nano, Mini, other 328-based: pin D2 (INT0) or D3 (INT1)
259 #define INT_PIN 2
260 
261 static void alarmHandler()
262 {
263  // Set global interrupt flag
264  alarmInterrupt = true;
265 }
266 
267 void setup()
268 {
269  ...
270 
271  // Attach to INT0 interrupt falling edge
272  pinMode(INT_PIN, INPUT_PULLUP);
273  attachInterrupt(digitalPinToInterrupt(INT_PIN), alarmHandler, FALLING);
274 
275  // Enable Alarm 1 and 2 interrupts
276  rtc.alarmInterruptEnable(Alarm1, true);
277  rtc.alarmInterruptEnable(Alarm2, true);
278 }
279 
280 void loop()
281 {
282  // Check global alarm interrupt flag
283  if (alarmInterrupt) {
284  if (rtc.getAlarmFlag(Alarm1)) {
285  // Handle alarm 1
286 
287  // Clear alarm 1 interrupt
288  rtc.clearAlarmFlag(Alarm1);
289  }
290 
291  if (rtc.getAlarmFlag(Alarm2)) {
292  // Handle alarm 2
293 
294  // Clear alarm 2 interrupt
295  rtc.clearAlarmFlag(Alarm2);
296  }
297  }
298 }
299 ```

306 rtc.outputClockPinEnable(true); // Enable
307 rtc.outputClockPinEnable(false); // Disable
