258  int8_t agingOffset;
259

262  agingOffset = regVal | ~((1 << 8) - 1);

271  ser->println(regVal & (1 << 7) ? F("1") : F("0"));
272 
273  ser->print(F(" [6:0]: 0x"));
274  ser->println(regVal, HEX);
275  break;
276  case DS3231_REG_TEMP_LSB:
277  ser->print(F(" [7:6]: 0x"));
278  ser->println(regVal & 0xC0, HEX);
279  break;
280  default:
281  ser->println(F("Unknown register"));
282  }

290 void DS3231Debug::printDiagnostics(Stream *ser)
291 {
292  bool oscillatorStopFlag;
293  bool alarm1Flag;
294  bool alarm2Flag;
295  int8_t agingOffset;
296  uint8_t status;
297  uint8_t control;
298 
299  ser->println(F("DS3231 diagnostics:"));
300 
301  oscillatorStopFlag = isOscillatorStopped();
302  agingOffset = getAgingOffset();
303  alarm1Flag = getAlarmFlag(Alarm1);
304  alarm2Flag = getAlarmFlag(Alarm2);
305  status = readStatusRegister();
306  control = readControlRegister();
307 
308  // ---------------------------------------------------------
309  ser->print(F(" I2C 7-bit address: 0x"));
310  ser->println(DS3231_ADDR, HEX);
311 
312  // ---------------------------------------------------------
313  ser->print(F(" Oscillator enabled: "));
314  ser->println((control & (1 << DS3231_CTRL_EOSC)) ? F("Stopped") : F("Started"));
315 
316  ser->print(F(" Oscillator stop flag: "));
317  ser->println(oscillatorStopFlag ? F("True") : F("False"));
318 
319  // ---------------------------------------------------------
320  ser->print(F(" 32kHz clock output: "));
321  ser->println(status & (1 << DS3231_STAT_EN32KHZ) ? F("Enable") : F("Disable"));
322 
323  ser->print(F(" INT/SQW mode: "));
324  ser->println((control & DS3231_CTRL_INTCN) ? F("Alarm interrupt") : F("SQW"));
325 
326  ser->print(F(" SQW: "));
327  if ((control & DS3231_CTRL_BBSQW) || (control & DS3231_CTRL_INTCN)) {
328  ser->println(F("Disabled"));
329  } else {
330  switch (control & 0x18) {
331  case 0x00:
332  ser->println(F("1Hz"));
333  break;
334  case 0x08:
335  ser->println(F("1024Hz"));
336  break;
337  case 0x10:
338  ser->println(F("4096Hz"));
339  break;
340  default:
341  ser->println(F("8192Hz"));
342  }
343  }
344 
345  // ---------------------------------------------------------
346  ser->print(F(" Alarm 1 interrupt: "));
347  ser->println((control & DS3231_CTRL_A1IE) ? F("Enable") : F("Disable"));
348 
349  ser->print(F(" Alarm 2 interrupt: "));
350  ser->println((control & DS3231_CTRL_A1IE) ? F("Enable") : F("Disable"));
351 
352  ser->print(F(" Alarm 1 flag: "));
353  ser->println(alarm1Flag);
354 
355  ser->print(F(" Alarm 2 flag: "));
356  ser->println(alarm2Flag);
357 
358  ser->print(F(" Temperature conversion busy: "));
359  ser->println(status & (1 << DS3231_STAT_BSY));
360 
361  // ---------------------------------------------------------
362  ser->print(F(" Aging offset: "));
363  ser->println(agingOffset, DEC);
364 }
