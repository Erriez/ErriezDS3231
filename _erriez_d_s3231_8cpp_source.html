71 bool ErriezDS3231::oscillatorEnable(bool enable)
72 {
73  uint8_t reg;
74 
75  // Read status register
76  reg = readRegister(DS3231_REG_CONTROL);
77 
78  // Set or clear EOSC bit in control register
79  if (enable) {
80  reg &= ~(1 << DS3231_CTRL_EOSC);
81  } else {
82  reg |= (1 << DS3231_CTRL_EOSC);
83  }
84 
85  // Write control register
86  if (!writeRegister(DS3231_REG_CONTROL, reg)) {
87  return false;
88  }
89 
90  // Clear OSF bit in status register
91  reg = readRegister(DS3231_REG_STATUS);
92  reg &= ~(1 << DS3231_STAT_OSF);
93  return writeRegister(DS3231_REG_STATUS, reg);
94 }

182 bool ErriezDS3231::read(struct tm *dt)
183 {
184  uint8_t buffer[7];
185 
186  // Read clock date and time registers
187  if (!readBuffer(0x00, buffer, sizeof(buffer))) {
188  memset(dt, 0, sizeof(struct tm));
189  return false;
190  }
191 
192  // Clear dt
193  memset(dt, 0, sizeof(struct tm));
194 
195  // Convert BCD buffer to Decimal
196  dt->tm_sec = bcdToDec(buffer[0] & 0x7F);
197  dt->tm_min = bcdToDec(buffer[1] & 0x7F);
198  dt->tm_hour = bcdToDec(buffer[2] & 0x3f);
199  dt->tm_wday = bcdToDec(buffer[3] & 0x07) - 1;
200  dt->tm_mday = bcdToDec(buffer[4] & 0x3F);
201  dt->tm_mon = bcdToDec(buffer[5] & 0x1f) - 1;
202  dt->tm_year = bcdToDec(buffer[6]) + 100;
203 
204  return true;
205 }

216 bool ErriezDS3231::write(const struct tm *dt)
217 {
218  uint8_t buffer[7];
219 
220  // Enable oscillator
221  if (!oscillatorEnable(true)) {
222  return false;
223  }
224 
225  // Encode date time from decimal to BCD
226  buffer[0] = decToBcd(dt->tm_sec) & 0x7F;
227  buffer[1] = decToBcd(dt->tm_min) & 0x7F;
228  buffer[2] = decToBcd(dt->tm_hour) & 0x3F;
229  buffer[3] = decToBcd(dt->tm_wday + 1) & 0x07;
230  buffer[4] = decToBcd(dt->tm_mday) & 0x3F;
231  buffer[5] = decToBcd(dt->tm_mon + 1) & 0x1F;
232  buffer[6] = decToBcd(dt->tm_year % 100);
233 
234  // Write BCD encoded buffer to RTC registers
235  return writeBuffer(0x00, buffer, sizeof(buffer));
236 }

375 bool ErriezDS3231::setAlarm1(Alarm1Type alarmType,
376  uint8_t dayDate, uint8_t hours, uint8_t minutes, uint8_t seconds)
377 {
378  uint8_t buffer[4];
379 
380  // Store alarm 1 registers in buffer
381  buffer[0] = decToBcd(seconds);
382  buffer[1] = decToBcd(minutes);
383  buffer[2] = decToBcd(hours);
384  buffer[3] = decToBcd(dayDate);
385 
386  // Set alarm 1 bits
387  if (alarmType & 0x01) { buffer[0] |= (1 << DS3231_A1M1); }
388  if (alarmType & 0x02) { buffer[1] |= (1 << DS3231_A1M2); }
389  if (alarmType & 0x04) { buffer[2] |= (1 << DS3231_A1M3); }
390  if (alarmType & 0x08) { buffer[3] |= (1 << DS3231_A1M4); }
391  if (alarmType & 0x10) { buffer[3] |= (1 << DS3231_DYDT); }
392 
393  // Write alarm 1 registers
394  if (!writeBuffer(DS3231_REG_ALARM1_SEC, buffer, sizeof(buffer))) {
395  return false;
396  }
397 
398  // Clear alarm 1 flag
399  return clearAlarmFlag(Alarm1);
400 }

617 {

619 
620  // Convert 8-bit signed value to register value
621  if (val < 0) {
622  // Calculate two's complement for negative value
623  regVal = ~(-val) + 1;

701 bool ErriezDS3231::getTemperature(int8_t *temperature, uint8_t *fraction)
702 {
703  uint8_t temp[2];
704 
705  // Read temperature MSB and LSB registers
706  if (!readBuffer(DS3231_REG_TEMP_MSB, &temp, sizeof(temp))) {
707  return false;
708  }
709 
710  // Set temperature argument
711  *temperature = temp[0];
712 
713  // Calculate two's complement when negative
714  if (*temperature & 0x80) {
715  *temperature |= ~((1 << 8) - 1);
716  }
717 
718  // Shift fraction bits 6 and 7 with 0.25 degree Celsius resolution
719  *fraction = (temp[1] >> 6) * 25;
720 
721  return true;
722 }

801 bool ErriezDS3231::writeBuffer(uint8_t reg, void *buffer, uint8_t len)
802 {
803  // Start I2C transfer by writing the I2C address, register number and optional buffer
804  Wire.beginTransmission(DS3231_ADDR);
805  Wire.write(reg);

810  return false;

812 
813  return true;

830 {
831  // Start I2C transfer by writing the I2C address and register number
832  Wire.beginTransmission(DS3231_ADDR);
833  Wire.write(reg);
834  // Generate a repeated start, followed by a read buffer
