141 void DS3231::setDateTime(DS3231_DateTime *dateTime)
142 {
143  uint8_t buffer[7];
144 
145  // Encode date time from decimal to BCD
146  buffer[0] = (decToBcd((uint8_t)(dateTime->second & 0x7F)));

152  buffer[6] = (decToBcd((uint8_t)((dateTime->year - 2000) & 0xFF)));
153 
154  // Write BCD encoded buffer to RTC registers

156

176 bool DS3231::getDateTime(DS3231_DateTime *dateTime)
177 {
178  uint8_t buf[7];
179 
180  // Read clock date and time registers
181  readBuffer(0x00, &buf, sizeof(buf));
182 
183  // Convert BCD buffer to Decimal
184  dateTime->second = bcdToDec(buf[0]);
185  dateTime->minute = bcdToDec(buf[1]);
186  dateTime->hour = bcdToDec(buf[2] & 0x3f);
187  dateTime->dayWeek = bcdToDec(buf[3]);
188  dateTime->dayMonth = bcdToDec(buf[4]);
189  dateTime->month = bcdToDec(buf[5] & 0x1f);
190  dateTime->year = 2000 + bcdToDec(buf[6]);
191 
192  // Check buffer for valid data
193  if ((dateTime->second > 59) ||
194  (dateTime->minute > 59) ||
195  (dateTime->hour > 23) ||
196  (dateTime->dayMonth < 1) || (dateTime->dayMonth > 31) ||
197  (dateTime->month < 1) || (dateTime->month > 12) ||
198  (dateTime->dayWeek < 1) || (dateTime->dayWeek > 7) ||
199  (dateTime->year > 2099))
200  {
201  // Invalid date/time read from RTC: Clear date time
202  memset(dateTime, 0x00, sizeof(DS3231_DateTime));
203  return true;
204  }
205 
206  return false;
207 }

284 uint32_t DS3231::getEpochTime(DS3231_DateTime *dateTime)
285 {
286  uint16_t year;
287  uint16_t days;
288  uint8_t i;
289 
290  // Convert date time to epoch
291  // Subtract year 2000
292  year = dateTime->year - 2000;
293 
294  // Calculate total number of days including leap days
295  days = ((365 * year) + (year + 3) / 4);
296 
297  // Add number of days each month in current year
298  for (i = 1; i < dateTime->month; i++) {
299  days += pgm_read_byte(daysMonth + i - 1);
300  }
301 
302  // Check month and leap year
303  if ((dateTime->month > 2) && ((year % 4) == 0)) {
304  days++;
305  }
306 
307  // Add number of days in current month
308  days += (uint16_t)(dateTime->dayMonth - 1);
309 
310  // Calculate epoch, starting at offset year 2000
311  return SECONDS_FROM_1970_TO_2000 +
312  (((((days * 24UL) + dateTime->hour) * 60) + dateTime->minute) * 60) + dateTime->second;
313 }

339 void DS3231::setAlarm1(Alarm1Type alarmType,
340  uint8_t dayDate, uint8_t hours, uint8_t minutes, uint8_t seconds)
341 {
342  uint8_t buffer[4];
343 
344  // Store alarm 1 registers in buffer
345  buffer[0] = decToBcd(seconds);
346  buffer[1] = decToBcd(minutes);
347  buffer[2] = decToBcd(hours);
348  buffer[3] = decToBcd(dayDate);
349 
350  // Set alarm 1 bits
351  if (alarmType & 0x01) { buffer[0] |= (1 << DS3231_A1M1); }
352  if (alarmType & 0x02) { buffer[1] |= (1 << DS3231_A1M2); }
353  if (alarmType & 0x04) { buffer[2] |= (1 << DS3231_A1M3); }
354  if (alarmType & 0x08) { buffer[3] |= (1 << DS3231_A1M4); }
355  if (alarmType & 0x10) { buffer[3] |= (1 << DS3231_DYDT); }
356 
357  // Write alarm 1 registers
358  writeBuffer(DS3231_REG_ALARM1_SEC, buffer, sizeof(buffer));
359 
360  // Clear alarm 1 flag
361  clearAlarmFlag(Alarm1);
362 }

633 void DS3231::getTemperature(int8_t *temperature, uint8_t *fraction)
634 {
635  uint8_t temp[2];
636 
637  // Read temperature MSB and LSB registers
638  readBuffer(DS3231_REG_TEMP_MSB, &temp, sizeof(temp));
639 
640  // Set temperature argument
641  *temperature = temp[0];
642 
643  // Calculate two's complement when negative
644  if (*temperature & 0x80) {
645  *temperature |= ~((1 << 8) - 1);
646  }
647 
648  // Shift fraction bits 6 and 7 with 0.25 degree Celsius resolution
649  *fraction = (temp[1] >> 6) * 25;
